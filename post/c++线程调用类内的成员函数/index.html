C++线程调用类内的成员函数

Posted by Liang Huimin on Monday, November 11, 2019

TOC

C++的线程在调用类的成员函数的不像调用普通的类外函数，需要在调用的时候传递额外的参数。

注意在给thread传递对象的时候按值传递、按地址传递和按引用传递的区别。

// ThreadCallMemberFunction.cpp : This file contains the 'main' function. Program execution begins and ends there.
//

#include "pch.h"
#include <iostream>
#include <thread>
using namespace std;
class Student {
public:
	Student(int num) {
		number_ = num;
	}
	void SetNumber(int i) {
		number_ = i;
	}
	void GetNumber() {
		cout << number_ << "\t" << this << endl;
	}
	void IncreaseNum(int i) {
		number_ += i;
		cout << number_ << "\t" << this << endl;	
	}
private:
	int number_ = 0;
};

int main()
{
	thread t1(&Student::GetNumber, Student(10));
	t1.detach();
	this_thread::sleep_for(chrono::milliseconds(100));
	Student st1(20);
	thread t2(&Student::IncreaseNum, st1, 5);
	t2.detach();
	this_thread::sleep_for(chrono::milliseconds(100));
	thread t3(&Student::IncreaseNum, st1, 5);
	t3.detach();
	this_thread::sleep_for(chrono::milliseconds(100));
	thread t4(&Student::IncreaseNum, &st1, 10);
	t4.detach();
	this_thread::sleep_for(chrono::milliseconds(100));
	thread t5(&Student::IncreaseNum, &st1, 10);
	t5.detach();
	this_thread::sleep_for(chrono::milliseconds(100));
	Student st2(100);
	thread t6(&Student::IncreaseNum, ref(st2), 5);
	t6.detach();
	thread t7(&Student::IncreaseNum, ref(st2), 5);
	t7.detach();
	this_thread::sleep_for(chrono::milliseconds(100));

	std::cout << "Finish!\n";
	this_thread::sleep_for(chrono::milliseconds(1000));
}
//输出结果
/*
10      00FFE978
25      00FFE8D4
25      00FFE8D4
30      00B5FE4C
40      00B5FE4C
105     00B5FE00
110     00B5FE00
Finish!
*/
